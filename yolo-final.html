Data preprocess¶

Output of the yolo is a 19*19 grid feature map. If the centre of the box is located in a grid, then this grid is responsible for containing the information about this box. For each grid, we prepared 5 anchor boxes.

def block1(inputs, layer_num = '1', output_num = 32, filter_size = 3):
    conv = Conv2D(filters = output_num,kernel_size = (filter_size,filter_size) ,strides = (1,1), padding = 'same',  use_bias=False,name = 'conv2d' + layer_num)(inputs)
    batch_norm = BatchNormalization(name = 'batch_norm' + layer_num)(conv)
    leaky_relu = LeakyReLU(name = 'leaky_relu' + layer_num)(batch_norm)
    maxpooling2d = MaxPooling2D(pool_size = (2,2), strides =(2,2), name = 'MaxPooling2D' + layer_num)(leaky_relu)
    return maxpooling2d

def block2(inputs, layer_num = '1',  output_num = 32, filter_size = 3):
    conv = Conv2D(filters = output_num,kernel_size = (filter_size,filter_size) ,strides = (1,1), padding = 'same', use_bias=False, name = 'conv2d' + layer_num)(inputs)
    batch_norm = BatchNormalization(name = 'batch_norm' + layer_num)(conv)
    leaky_relu = LeakyReLU(name = 'leaky_relu' + layer_num)(batch_norm)
    return leaky_relu

def cilp_output(input):
    coords_confidence =  input[..., :5]
    car_class = K.expand_dims(input[..., 7], axis=-1)
    truck_class = K.expand_dims(input[..., 12], axis=-1)
    output = K.concatenate((coords_confidence,car_class,truck_class), axis = -1)
    return output

def loss_function(y_true_origin, y_pred_origin):

    y_true = tf.reshape(y_true_origin, (-1,19,19,5,7))
    y_pred = tf.reshape(y_pred_origin, (-1,19,19,5,7))


    pred_box_conf, pred_box_xy, pred_box_wh, pred_box_class = yolo_head_loss(y_pred)


    true_box_conf =  y_true[..., 4]  
    true_box_xy = y_true[..., :2] 
    true_box_wh = y_true[..., 2:4]
    true_box_class = y_true[..., 5:]
    
    ### Calculate the IOU for grid with obeject
    true_wh_half = true_box_wh / 2.
    true_mins    = true_box_xy - true_wh_half
    true_maxes   = true_box_xy + true_wh_half
    
    pred_wh_half = pred_box_wh / 2.
    pred_mins    = pred_box_xy - pred_wh_half
    pred_maxes   = pred_box_xy + pred_wh_half       
    
    intersect_mins  = tf.maximum(pred_mins,  true_mins)
    intersect_maxes = tf.minimum(pred_maxes, true_maxes)
    intersect_wh    = tf.maximum(intersect_maxes - intersect_mins, 0.)
    intersect_areas = intersect_wh[..., 0] * intersect_wh[..., 1]
    
    true_areas = true_box_wh[..., 0] * true_box_wh[..., 1]
    pred_areas = pred_box_wh[..., 0] * pred_box_wh[..., 1]
    union_areas = pred_areas + true_areas - intersect_areas
    iou_scores_obj = tf.truediv(intersect_areas, union_areas)  
    
    best_ious_obj = tf.reduce_max(iou_scores_obj, axis=-1, keep_dims=True)   # Best IOU scores.

    object_detections = K.cast(best_ious_obj > 0.6, K.dtype(best_ious_obj))

    
    ### class mask: simply the position of the ground truth boxes (the predictors) 
    class_mask = true_box_conf

    ### coordinate mask: simply the position of the ground truth boxes 
    coord_mask = tf.expand_dims(true_box_conf, axis=-1)
    
    ### cofident mask: 
    # for grid without object, penelize if ious with anchor box<0.6
    conf_mask_noobj =  (1 - true_box_conf) * (1 - object_detections)
    conf_mask_noobj = tf.expand_dims(conf_mask_noobj,axis = -1)  
    conf_mask_obj =  tf.expand_dims(true_box_conf,axis = -1)  

        
    nb_coord_box = tf.reduce_sum(tf.to_float(coord_mask > 0.0))
    nb_conf_noobj_box  = tf.reduce_sum(tf.to_float(conf_mask_noobj  > 0.0))
    nb_coord_obj_box = tf.reduce_sum(tf.to_float(conf_mask_obj  > 0.0))
    nb_class_box = tf.reduce_sum(tf.to_float(class_mask > 0.0))
    
    loss_xy = tf.reduce_sum(tf.square(true_box_xy-pred_box_xy)  * coord_mask) / (nb_coord_box + 1e-6) / 2.
    loss_wh = tf.reduce_sum(tf.square(tf.sqrt(true_box_wh)-tf.sqrt(pred_box_wh))  * coord_mask) / (nb_coord_box + 1e-6) / 2.
    loss_coord = 5 * loss_xy + 5 * loss_wh
    
    true_box_conf = tf.expand_dims(true_box_conf,axis = -1)
    iou_scores_obj = tf.expand_dims(iou_scores_obj,axis = -1)
    true_box_conf_adjust = true_box_conf * iou_scores_obj #for grid with object, we hope target confidence = iou *1
    
    loss_conf_obj = tf.reduce_sum(tf.square(true_box_conf_adjust-pred_box_conf) * conf_mask_obj)  / (nb_coord_obj_box  + 1e-6) / 2.
    loss_conf_noobj = tf.reduce_sum(tf.square(true_box_conf-pred_box_conf) * conf_mask_noobj)  / (nb_conf_noobj_box  + 1e-6) / 2.
    loss_conf = loss_conf_obj + 0.1 * loss_conf_noobj

    
    loss_class = tf.nn.softmax_cross_entropy_with_logits(labels=true_box_class, logits=pred_box_class)
    loss_class = tf.reduce_sum(loss_class) / (nb_class_box + 1e-6)
    loss_class = tf.reduce_sum(loss_class * class_mask) / (nb_class_box + 1e-6)
    
    #class_mask = tf.expand_dims(class_mask, axis=-1)
    #loss_class = tf.square(true_box_class-pred_box_class)
    #loss_class = tf.reduce_sum(loss_class * class_mask) / (nb_class_box + 1e-6)
    
    #According to the yolo paper, a coeffiicent is applied to diffent part of the loss so that the model training is more stable
    loss = loss_coord + loss_conf + loss_class

    return loss

def process_image(img):    
    yolo_outputs = yolo_head(yolo_model.output, anchors, 80)
    scores, boxes, classes = yolo_eval(yolo_outputs, image_shape)
    resized_image = cv2.resize(img, (608,608))
    image_data = np.array(resized_image, dtype='float32')
    image_data /= 255.
    image_data = np.expand_dims(image_data, 0)  
    out_scores, out_boxes, out_classes =sess.run([scores, boxes, classes],feed_dict={yolo_model.input:image_data, K.learning_phase(): 0})
    img = draw_box(img, out_boxes, color=(0, 0, 255), thick=6, image_shape = [1920,1200])
    return img
